## Components and Interfaces

### 1. Station Data Structure

```typescript
interface Station {
  id: string;
  name: string;
  nameVi: string;
  icon: string;           // Emoji icon
  description: string;
  descriptionVi: string;
  color: string;          // Gradient colors
  bgColor: string;        // Background color
  tools: Tool[];
}

interface Tool {
  id: string;
  name: string;
  nameVi: string;
  description: string;
  descriptionVi: string;
  icon: string;           // Lucide icon name or emoji
  creditCost: number;
  tier: 'free' | 'plus' | 'pro';
  estimatedTime: number;  // seconds
  inputType: 'image' | 'video' | 'both';
  options?: ToolOption[];
}

interface ToolOption {
  id: string;
  label: string;
  labelVi: string;
  type: 'select' | 'slider' | 'toggle' | 'text';
  values?: string[];
  min?: number;
  max?: number;
  default: any;
}
```
